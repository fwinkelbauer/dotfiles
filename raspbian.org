** Updates

#+begin_src python :tangle "~/.local/bin/my-upgrade" :shebang "#!/bin/python3"
import subprocess
import os


def announce(msg):
    print()
    print(msg)
    print("========================================")


def run(args, cwd=None):
    subprocess.run(args, check=True, cwd=cwd)


def apt():
    run(["sudo", "apt", "update"])
    run(["sudo", "apt", "upgrade", "-y"])
    run(["sudo", "apt", "auto-remove", "-y"])


def git_babel(directory, tangle):
    directory = os.path.expanduser(directory)
    tangle = os.path.join(directory, tangle)
    run(["git", "pull"], directory)
    run(["emacs", "-Q", "--batch", "-l", "org", "--eval", f'(org-babel-tangle-file "{tangle}")'])


def main():
    announce("Install updates")
    apt()
    announce("Update dotfiles")
    git_babel("~/.config/emacs", "raspbian.org")


if __name__ == '__main__':
    main()
#+end_src
