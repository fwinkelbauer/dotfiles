#+begin_src sh :tangle "~/.local/bin/my-copy" :shebang "#!/bin/bash"
set -euo pipefail

local=~/.chunkyard

remotes=(
    /media/florian/Backup1/Florian/main.chunkyard
    /media/florian/Backup2/Florian/main.chunkyard
)

echo "Cleaning up: $local"
chunkyard keep -r "$local" --latest 12 --prompt Libsecret
chunkyard gc -r "$local" --prompt Libsecret

for remote in "${remotes[@]}"; do
    if [ -d "$remote" ]; then
        echo "Pushing to: $remote"
        chunkyard copy -r "$local" -d "$remote" --prompt Libsecret

        echo "Pulling from: $remote"
        chunkyard copy -r "$remote" -d "$local" --prompt Libsecret

        echo "Cleaning up: $remote"
        chunkyard keep -r "$remote" --latest 12 --prompt Libsecret
        chunkyard gc -r "$remote" --prompt Libsecret
    fi
done
#+end_src

** Homebank

#+begin_src sh :tangle "~/.local/bin/my-homebank" :shebang "#!/bin/bash"
set -euo pipefail

input=$(find ~/Downloads -name 'ums*.csv')
output=~/Downloads/homebank.csv

iconv -f cp1252 -t utf8 "$input" | awk '{ FS=";"; RS="\r?\n"; gsub("\"", "", $10); if (NR!=1) { print $3 ";8;;;" $10 ";" $8 ";;" } }' > "$output"
rm "$input"
#+end_src
